\subsection{Typing Rules}
\label{subsec:axcut:typing-rules}

\begin{prooftree}
    \AxiomC{$\Gamma \vdash s_1 \dashv \Gamma'$}
    \AxiomC{$\Gamma' \vdash s_2 \dashv \Gamma''$}
    \RightLabel{\textsc{Seq}}
    \BinaryInfC{$\Gamma \vdash s_1 ; s_2 \dashv \Gamma''$}
\end{prooftree}

\begin{prooftree}
    \AxiomC{$\sigma : \Gamma \Rightarrow \Gamma'$}
    \RightLabel{\textsc{Subst}}
    \UnaryInfC{$\Gamma \vdash \mathbf{substitute}(\sigma) \dashv \Gamma'$}
\end{prooftree}
